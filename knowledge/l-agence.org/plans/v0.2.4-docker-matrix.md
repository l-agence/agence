**Matrix Math Foundation**:
```typescript
priority(task) = 1 + count(tasks_blocked_by_this)
complexity(task) = eval(loc, modules_affected)
recommended_model(priority, complexity) = routing_table[pri_tier][cplx_tier]

// Merge conflict resolution
function resolveConflict(path, theirs, ours, ancestor) {
  const agentA = getMetadata(getAuthor(theirs));
  const agentB = getMetadata(getAuthor(ours));
  
  if (agentA.security_label === 'critical') {
    return MANUAL_REVIEW;  // No auto-merge for security
  }
  return agentA.priority > agentB.priority ? theirs : ours;
}
```
